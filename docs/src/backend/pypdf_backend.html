 43    def merge(self,
 44              output_file: str,
 45              input_files: dict[str, str],
 46              ):
 47        """
 48        Merge input files into an output.
 49
 50        ```python
 51        input_files = {
 52            file_path_1: password_1,
 53            file_path_2: password_2,
 54            ...
 55        }
 56        ```
 57
 58        :param output_file: Output PDF file.
 59        :param input_files: Input PDF files. Format ``dict[file_path, password]`` (if file not encrypted, use ``password = None``).
 60
 61        :raises FileNotFoundError: if input file not found.
 62        """
 63        output_file = output_file.replace(".pdf", "")
 64        with PdfWriter() as writer:
 65            for in_file, decrypt_password in input_files.items():
 66                self.check_file_exists(in_file)
 67                with PdfReader(in_file) as reader:
 68                    if decrypt_password and reader.is_encrypted:
 69                        reader.decrypt(decrypt_password)
 70                    for page in reader.pages:
 71                        writer.add_page(page)
 72            writer.write(f"{output_file}.pdf")

 96                    writer.add_page(page)

126    def rotate(self,
127               output_file: str,
128               input_file: str,
129               rotations: dict[int, int],
130               decrypt_password: str | None = None,
131               ):
132        """
133        Rotate specific pages from input files, generating a PDF output file.
134
135        :param output_file: Output PDF file.
136        :param input_file: Input PDF file.
137
138        :param rotations: Dict format { page_num: rotation_degrees }. Degree must be 0 or multiples of 90 degrees - positive or negative.
139        :param decrypt_password: Password used to decrypt file, if needed. Defaults to None (do not decrypt).
140
141        :raises FileNotFoundError: if input file not found
142        :raises ValueError: if rotation degree is invalid (valid values are 0 or multiples of 90 degrees - positive or negative).        
143        """
144        output_file = output_file.replace(".pdf", "")
145        self.check_file_exists(input_file)
146        with PdfReader(input_file) as reader, PdfWriter() as writer:
147            if decrypt_password and reader.is_encrypted:
148                reader.decrypt(decrypt_password)
149
150            for i, page in enumerate(reader.pages):
151                rotation = int(rotations.get(i, 0))
152
153                # parse rotation argument
154                rotation = int(math.fmod(rotation, 360))
155                if rotation < 0:
156                    rotation += 360  # fix rotation signal
157
158                if rotation not in (0, 90, 180, 270):
159                    raise ValueError(f"Page '{i}' rotation {rotation} is invalid. Rotation must be 0, 90, 180 or 270 degrees.")
160
161                # execute page rotation
162                print("Instruction:", i, rotation)
163                if rotation > 0:
164                    page.rotate(rotation)  # clockwise: 90, 180, 270
165                writer.add_page(page)
166            # save output file
167            writer.write(f"{output_file}.pdf")

169    def encrypt(self,
170                output_file: str,
171                input_file: str,
172                owner_password: str,
173                user_password: str | None = None,
174                decrypt_password: str | None = None,
175                permission_annotate: bool = False,
176                permission_fill_forms: bool = False,
177                permission_modify: bool = False,
178                permission_modify_pages: bool = False,
179                permission_copy: bool = False,
180                permission_accessibility: bool = True,
181                permission_print_low_quality: bool = True,
182                permission_print_high_quality: bool = True,
183                permission_all: bool = False,
184                encryption_algorithm: str = "AES-256",
185                ):
186        """
187        Encrypt input file, generating a protected PDF output file.
188
189        **Available permissions:**
190        - ``permission_annotate``: User can add/modify annotations (comments, highlight text, etc) and interactive forms. Default to False (not set).
191        - ``permission_fill_forms``: User can fill form fields (subset permission of `permission_annotate`). Default to False (not set).
192        - ``permission_modify``: User can modify the document (e.g., add / edit text, add / edit images, etc). Default to False (not set).
193        - ``permission_modify_pages``: User can insert, delete, or rotate pages (subset of ``permission_modify``). Default to False (not set).
194        - ``permission_copy``: User can copy text/images. Default to False (not set).
195        - ``permission_accessibility``: User can use screen readers for accessibility. Default to True (not set).
196        - ``permission_print_low_quality``: User can PRINT pdf (low quality). Defaults to True (allow).
197        - ``permission_print_high_quality``: User can PRINT pdf (high quality). Requires `permission_print_low_quality=True`. Defaults to True (allow).
198        - ``permission_all``: User has ALL PERMISSIONS. If True, it overrides all other permissions. Defaults to False (not set).
199
200        **Encryption algorithms:**
201        - ``AES-256``    (high security, compatible with 2008+ PDF readers - **RECOMMENDED**)
202        - ``AES-128``    (good security, good compatibility - **RECOMMENDED**)
203        - ``AES-256-R5`` (high security,  lowest compatibility - not recommended)
204        - ``RC4-128``    (weak security, highest compatibility - not recommended)
205        - ``RC4-40``     (weakest security - not recommended)
206
207        :param output_file: Output PDF file.
208        :param input_file: Input PDF file.
209
210        :param owner_password: Owner password for encryption. Owner has ALL PERMISSIONS in the output PDF file.
211        :param user_password:  User password for encryption. User has ONLY THE PERMISSIONS specified in the arguments. Defaults to None (user and owner password are the same).
212        :param decrypt_password: Password used to decrypt file, if needed. Defaults to None (do not decrypt).
213
214        :param permission_annotate: User can add/modify annotations (comments, highlight text, etc) and interactive forms. Default to False (not set).
215        :param permission_fill_forms: User can fill form fields (subset permission of `permission_annotate`). Default to False (not set).
216        :param permission_modify: User can modify the document (e.g., add / edit text, add / edit images, etc). Default to False (not set).
217        :param permission_modify_pages: User can insert, delete, or rotate pages (subset of ``permission_modify``). Default to False (not set).
218        :param permission_copy: User can copy text/images. Default to False (not set).
219        :param permission_accessibility: User can use screen readers for accessibility. Default to True (allow).
220        :param permission_print_low_quality: User can PRINT pdf (low quality). Defaults to True (allow).
221        :param permission_print_high_quality: User can PRINT pdf (high quality). Requires `permission_print_low_quality=True`. Defaults to True (allow).
222        :param permission_all: User has ALL PERMISSIONS. If True, it overrides all other permissions. Defaults to False (not set).
223
224        :param encryption_algorithm: Encryption algorithm used. Valid options are "RC4-40", "RC4-128", "AES-128", "AES-256-R5", or "AES-256". Defaults to "AES-256" (for enhanced security and compatibility).
225
226        :raises FileNotFoundError: if input file not found
227        """
228        output_file = output_file.replace(".pdf", "")
229        self.check_file_exists(input_file)
230        with PdfReader(input_file) as reader, PdfWriter() as writer:
231            if decrypt_password and reader.is_encrypted:
232                reader.decrypt(decrypt_password)
233
234            if encryption_algorithm not in (None, "RC4-40", "RC4-128", "AES-128", "AES-256-R5", "AES-256"):
235                raise ValueError(f"Encryption algorithm '{encryption_algorithm}' is invalid.  Valid options are 'RC4-40', 'RC4-128', 'AES-128', 'AES-256-R5', or 'AES-256'.")
236
237            # set user permissions
238            permissions = UserAccessPermissions(0)
239            if permission_annotate:
240                permissions |= UserAccessPermissions.ADD_OR_MODIFY
241            if permission_fill_forms:
242                permissions |= UserAccessPermissions.FILL_FORM_FIELDS
243            if permission_modify:
244                permissions |= UserAccessPermissions.MODIFY
245            if permission_modify_pages:
246                permissions |= UserAccessPermissions.ASSEMBLE_DOC
247            if permission_copy:
248                permissions |= UserAccessPermissions.EXTRACT
249            if permission_accessibility:
250                permissions |= UserAccessPermissions.EXTRACT_TEXT_AND_GRAPHICS
251            if permission_print_low_quality:
252                permissions |= UserAccessPermissions.PRINT
253            if permission_print_high_quality:
254                permissions |= UserAccessPermissions.PRINT_TO_REPRESENTATION
255            if permission_all:
256                permissions |= UserAccessPermissions.all()
257
258            # add pages into writer
259            for page in reader.pages:
260                writer.add_page(page)
261
262            # Encrypt with user and owner passwords
263            writer.encrypt(
264                owner_password=owner_password,
265                user_password=user_password if user_password else owner_password,
266                permissions_flag=permissions,
267                algorithm=encryption_algorithm,
268                use_128bit=True,
269            )
270
271            # save output file
272            writer.write(f"{output_file}.pdf")
